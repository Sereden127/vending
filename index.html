import express from "express"; import crypto from "crypto"; const app = express(); app.use(express.json()); // 🔐 CONFIGURACIÓN (rellena con tus datos reales) const SECRET_KEY = "TU_CLAVE_SECRETA_BASE64"; const MERCHANT_CODE = "367445079"; const TERMINAL = "001"; const CURRENCY = "978"; const URL_BASE = "https://fersomaticweb.onrender.com/pago-ok-seleccion"; // 🧠 generar order único function generateOrder() { return "FER" + Date.now().toString().slice(-10); } // 🔐 firma HMAC SHA256 (Redsys style) function sign(data, key) { const keyDecoded = Buffer.from(key, "base64"); return crypto .createHmac("sha256", keyDecoded) .update(data) .digest("base64"); } app.post("/api/crear-pedido", (req, res) => { const { seleccion, email, sugarLevel, machine } = req.body; const order = generateOrder(); const importe = (machine === "snack") ? "100" : "50"; // 📦 merchant params const merchantParams = { Ds_Merchant_Amount: importe, Ds_Merchant_Currency: CURRENCY, Ds_Merchant_Order: order, Ds_Merchant_MerchantCode: MERCHANT_CODE, Ds_Merchant_Terminal: TERMINAL, Ds_Merchant_TransactionType: "0", Ds_Merchant_MerchantData: "", Ds_Merchant_UrlOK: "", Ds_Merchant_UrlKO: "", Ds_Merchant_Product: seleccion, Ds_Merchant_ConsumerLanguage: "1" }; const merchantBase64 = Buffer.from(JSON.stringify(merchantParams)).toString("base64"); const signature = sign(merchantBase64, SECRET_KEY); // 📦 datos extra tuyos const datos = { seleccion, userEmail: email, sugarLevel: sugarLevel || 0, numeroPedido: order, machine }; const datosBase64 = Buffer.from(JSON.stringify(datos)).toString("base64"); const url = `${URL_BASE}?datos=${datosBase64}` + `&Ds_MerchantParameters=${merchantBase64}` + `&Ds_SignatureVersion=HMAC_SHA256_V1` + `&Ds_Signature=${encodeURIComponent(signature)}`; res.json({ url }); }); app.listen(3000, () => { console.log("Servidor vending activo en puerto 3000"); });
